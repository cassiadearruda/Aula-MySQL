--CRIAR TABELA DE PRODUTOS COM 5 "ATRIBUTOS"
use db_ecommerce;
create table tb_produtos (
id bigint (3) auto_increment,
nome varchar (50) not null,
marca varchar (50) not null,
preco bigint (5),
estoque bigint (3),
primary key (id)
);
