create database db_escola;

--CRIAR TABELA DE ESTUDANTES COM 5 "ATRIBUTOS"
use db_escola;
create table tb_estudantes (
id bigint (3) auto_increment,
nome varchar (50) not null,
idade bigint (2),
nota decimal (2),
ativo boolean,
primary key (id)
);
